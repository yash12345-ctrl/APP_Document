Building the Journey App

This documentation outlines the complete process of building the Journey Flutter application. It includes step-by-step instructions on setting up assets, configuring dependencies, and the full source code for the complex animations and Text-to-Speech (TTS) integrations we implemented.

Advanced Animations

Custom AnimationController logic for flying, bobbing, and pulsing effects.

Text-to-Speech

Integrated flutter_tts to provide audio feedback on every screen transition.

Layered UI

Complex Stack layouts with gradients, SVGs, and custom clippers.

1 Assets & Dependencies

First, update your pubspec.yaml to include the TTS package and register your image assets.

pubspec.yaml 
dependencies:
  flutter:
    sdk: flutter
  cupertino_icons: ^1.0.8
  flutter_tts: ^3.8.3  # Required for audio

flutter:
  uses-material-design: true
  assets:
    - assets/pirate_frog.png
    - assets/guru_moksh.png
    - assets/pirate_man.png
    - assets/frog_in_boat.png
    - assets/speech_bubble.png
    - assets/waves.png

Important: Ensure you create an assets folder in your project root and place all the required PNG images inside it before running the app.

2 Welcome Screen (main.dart)

This acts as the entry point. It features the "AHOY!" waving text and the floating Pirate Frog. It speaks "Welcome aboard!" upon entry.

lib/main.dart 
import 'package:flutter/material.dart';
import 'package:flutter/material.dart';
import 'dart:math' as math;
import 'package:flutter_tts/flutter_tts.dart';
import 'moksh_screen.dart';

void main() {
  runApp(const HoppieApp());
}

class HoppieApp extends StatelessWidget {
  const HoppieApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Hoppie UI Demo',
      debugShowCheckedModeBanner: false,
      theme: ThemeData(
        fontFamily: 'Roboto',
        useMaterial3: true,
      ),
      home: const HoppieWelcomeScreen(),
    );
  }
}

class HoppieWelcomeScreen extends StatefulWidget {
  const HoppieWelcomeScreen({super.key});

  @override
  State createState() => _HoppieWelcomeScreenState();
}

class _HoppieWelcomeScreenState extends State with TickerProviderStateMixin {
  static const Color _darkBlueTop = Color(0xFF050914);
  static const Color _darkBlueBottom = Color(0xFF0B152E);
  static const Color _themeGreen = Color(0xFF387C24);
  static const Color _buttonBorderColor = Color(0xFF03070E);

  late AnimationController _entryController;
  late AnimationController _floatController;
  late AnimationController _pulseController;

  late Animation _fadeAnimation;
  late Animation _slideAnimation;
  late Animation _pulseAnimation;

  final FlutterTts flutterTts = FlutterTts();

  @override
  void initState() {
    super.initState();

    _entryController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1200),
    );

    _fadeAnimation = CurvedAnimation(
      parent: _entryController,
      curve: Curves.easeOut,
    );

    _slideAnimation = Tween(
      begin: const Offset(0, 0.1),
      end: Offset.zero,
    ).animate(CurvedAnimation(
      parent: _entryController,
      curve: Curves.easeOutQuart,
    ));

    _entryController.forward();

    _floatController = AnimationController(
      vsync: this,
      duration: const Duration(seconds: 3),
    )..repeat(reverse: false);

    _pulseController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1500),
    )..repeat(reverse: true);

    _pulseAnimation = Tween(begin: 1.0, end: 1.1).animate(
      CurvedAnimation(parent: _pulseController, curve: Curves.easeInOut),
    );

    _speakWelcome();
  }

  Future _speakWelcome() async {
    await flutterTts.setLanguage("en-US");
    await flutterTts.setPitch(1.0);
    await flutterTts.setSpeechRate(0.7);
    await flutterTts.speak("Welcome aboard!");
  }

  @override
  void dispose() {
    _entryController.dispose();
    _floatController.dispose();
    _pulseController.dispose();
    flutterTts.stop();
    super.dispose();
  }

  Widget _buildWavyText(String text, TextStyle style) {
    return Row(
      mainAxisSize: MainAxisSize.min,
      mainAxisAlignment: MainAxisAlignment.center,
      children: text.split('').asMap().entries.map((entry) {
        int index = entry.key;
        String char = entry.value;
        
        return AnimatedBuilder(
          animation: _floatController,
          builder: (context, child) {
            double offsetY = 6.0 * math.sin((_floatController.value * 2 * math.pi) + (index * 0.5));
            return Transform.translate(
              offset: Offset(0, offsetY),
              child: Text(char, style: style),
            );
          },
        );
      }).toList(),
    );
  }

  @override
  Widget build(BuildContext context) {

    return Scaffold(
      body: Stack(

          Container(
            decoration: const BoxDecoration(
              gradient: LinearGradient(
                begin: Alignment.topCenter,
                end: Alignment.bottomCenter,
                colors: [_darkBlueTop, _darkBlueBottom],
                stops: [0.0, 0.7],
              ),
            ),
          ),

          Positioned.fill(
            bottom: 100,
            child: FadeTransition(
              opacity: _fadeAnimation,
              child: SlideTransition(
                position: _slideAnimation,
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    _buildWavyText(
                      'AHOY!',
                      const TextStyle(
                        color: Colors.white,
                        fontSize: 40,
                        fontWeight: FontWeight.w900,
                        letterSpacing: 3.0,
                      ),
                    ),
                    
                    const SizedBox(height: 16),
                    const Text(
                      'Welcome aboard!',
                      style: TextStyle(
                        color: Colors.white,
                        fontSize: 24,
                        fontWeight: FontWeight.w700,
                      ),
                    ),
                    const SizedBox(height: 40),
                    const Text(
                      'Learn Language as you sail\nthrough islands of\nadventure!',
                      textAlign: TextAlign.center,
                      style: TextStyle(
                        color: Colors.white,
                        fontSize: 18,
                        height: 1.5,
                        fontWeight: FontWeight.w400,
                      ),
                    ),
                    const SizedBox(height: 40),

                    AnimatedBuilder(
                      animation: _floatController,
                      builder: (context, child) {
                        return Transform.translate(
                          offset: Offset(0, 10 * math.sin(_floatController.value * 2 * math.pi)),
                          child: child,
                        );
                      },
                      child: Image.asset(
                        'assets/pirate_frog.png',
                        height: 180,
                        fit: BoxFit.contain,
                      ),
                    ),

                    const SizedBox(height: 30),
                    const Text(
                      'HOPPIE',
                      style: TextStyle(
                        color: Colors.white,
                        fontSize: 18,
                        fontWeight: FontWeight.w500,
                        letterSpacing: 1.2,

class BottomHumpClipper extends CustomClipper {
  @override
  Path getClip(Size size) {
    final path = Path();
    path.moveTo(0, size.height);
    path.lineTo(0, size.height * 0.4);

    path.quadraticBezierTo(
        size.width / 2,
        -size.height * 0.6,
        size.width,
        size.height * 0.4
        );

    path.lineTo(size.width, size.height);
    path.close();
    return path;
  }

  @override
  bool shouldReclip(CustomClipper oldClipper) => false;
}

class BottomHumpClipper extends CustomClipper {
  @override
  Path getClip(Size size) {
    final path = Path();
    path.moveTo(0, size.height);
    path.lineTo(0, size.height * 0.4);
    path.quadraticBezierTo(
        size.width / 2, -size.height * 0.6, size.width, size.height * 0.4);
    path.lineTo(size.width, size.height);
    path.close();
    return path;
  }
  @override
  bool shouldReclip(CustomClipper oldClipper) => false;
}
